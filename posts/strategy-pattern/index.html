from abc import ABC, abstractmethod


class Player(ABC):
    def __init__(self, name: str, country: str):
        self.name = name
        self.country = country    

    @abstractmethod
    def bat(self) -> None:
        raise NotImplementedError

    @abstractmethod
    def bowl(self) -> None:
        raise NotImplementedError

    def __str__(self) -> str:
        return f"My name is {self.name}. I'm from {self.country}."
